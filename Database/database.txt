patients:{
    id(auto increments)
    first name
    last name
    id_number
    age
    phone_number
    gender
    conditions
    status(treated/not treated)
}

doctors:{
    id(auto increment)
    first name
    last name
    username
    speciality
    open_availability(date, time)
    close_availability(date,time)
    availability(enum(true, false)  - available or not) - depends on the  date and time.
}
medicines:{
    id(auto increment)
    name
    measurements(grams)
    count
}
appointments:{
    id(auto increment)
    patient_id (foreign key from patients)
    doctor_id  (foreign key from doctors)
}
